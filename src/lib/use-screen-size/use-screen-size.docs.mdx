import useScreenSize from '.'

export const meta = {
  type: 'hook',
  name: 'useScreenSize',
  description: 'Hook for conditional rendering content on different viewports',
  tags: ['hook'],
  lists: {
    arguments: [],
    returns: [
      { name: 'width', type: 'number', description: 'Width of viewport' },
      { name: 'height', type: 'number', description: 'Height of viewport' },
      { name: 'lessThanSmall', type: 'boolean', description: 'Is width less than 576px' },
      { name: 'sizeS', type: 'boolean', description: 'Is width more than 375px' },
      { name: 'greaterThanSmall', type: 'boolean', description: 'Is width more than 576px' },
      { name: 'greaterThanMedium', type: 'boolean', description: 'Is width more than 768px' },
      { name: 'greaterThanLarge', type: 'boolean', description: 'Is width more than 992px' },
      { name: 'greaterThanExtraLarge', type: 'boolean', description: 'Is width more than 1200px' },
    ],
  },
  scope: { useScreenSize },
}

## Playground

```jsx live render
const ScreenSize = () => {
  const screenSize = useScreenSize()
  const getStyle = (isTruthy) => isTruthy ? {color: 'green'} : {color: 'red'}

  return <>
    <h3>Resize the page to see hook in action</h3>
    <p>Current width  - {screenSize.width}px</p>
    <p>Current height - {screenSize.height}px</p>
    <br/>
    <p style={getStyle(screenSize.lessThanSmall)}><b>lessThanSmall</b> - width less than 576px</p>
    <p style={getStyle(screenSize.sizeS)}><b>SizeS</b> - width more than 375px</p>
    <p style={getStyle(screenSize.greaterThanSmall)}><b>greaterThanSmall</b> - width more than 576px</p>
    <p style={getStyle(screenSize.greaterThanMedium)}><b>greaterThanMedium</b> - width more than 768px</p>
    <p style={getStyle(screenSize.greaterThanLarge)}><b>greaterThanLarge</b> - width more than 992px</p>
    <p style={getStyle(screenSize.greaterThanExtraLarge)}><b>greaterThanExtraLarge</b> - width more than 1200px</p>
  </>
}

render(<ScreenSize />)
```
