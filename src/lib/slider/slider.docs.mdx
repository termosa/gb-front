const CarouselPreview = () => {
  const [isScrollbarPresent, setScrollbarPresent] = React.useState()
  const [isDotsPresent, setDotsPresent] = React.useState()
  const [isPartiallyVisible, setPartiallyVisible] = React.useState(true)
  const sliderSettings = {
    desktop: {
       breakpoint: { max: 3000, min: 768 },
       items: isPartiallyVisible ? 3 : 1,
    },
    tablet: {
       breakpoint: { max: 768, min: 464 },
       items: isPartiallyVisible ? 2.5 : 1,
    },
    mobile: {
       breakpoint: { max: 464, min: 0 },
       items: isPartiallyVisible ? 1.5 : 1,
    },
  }
  return (
  <Slider
    responsive={sliderSettings}
    scrollbarPresent={isScrollbarPresent}
    dotsPresent={isDotsPresent}
  >
     {[...new Array(5).fill()].map((_, i) => {
        return <Image src={"https://source.unsplash.com/random/800x800/?img=" + i} alt="" style={{width: "300px"}}/>
     })}
  </Slider>
  )
}

render(<CarouselPreview />)
