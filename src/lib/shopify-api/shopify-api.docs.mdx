## Usage

```js
import shopifyApi from './shopify-api'

// Request cart information
shopifyApi({ path: '/cart.js' })
  .then(cart => alert(`${cart.item_count} item(s) in the cart`))

// Add product to the cart
shopifyApi({
  path: '/cart/add.js',
  method: 'POST',
  body: {
    items: [{
      id: 36110175633573,
      quantity: 2,
    }],
  },
})
```
