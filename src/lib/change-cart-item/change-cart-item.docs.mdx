## Usage

```js
import changeCartItem from './change-cart-item'
changeCartItem('794864229', { discount: 0.6 }, 5).then(r => {
  const changedCartItem = r.find(item => item.id === '794864229')
  console.log(`${changedCartItem.title} quantity now is equal to ${changedCartItem.quantity}`)
})
```
