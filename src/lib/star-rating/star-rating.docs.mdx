## Playground

```jsx live render
const StarRatingExample = () => {
const [reviewAverage, setReviewAverage] = useState(2.4)
const [reviewCount, setReviewCount] = useState(32)

  return (
    <div>
      <input type='number' value={reviewAverage} onChange={e => setReviewAverage(+e.target.value)} />
      <input value={reviewCount} onChange={e => setReviewCount(+e.target.value)} />
     <StarRating reviewAverage={reviewAverage} reviewCount={reviewCount} />
    </div>
  )
}

render(<StarRatingExample />)
```
