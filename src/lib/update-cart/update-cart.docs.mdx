## Usage

```js
import updateCart from './update-cart'
updateCart({
  note: 'This is a note about my order',
  attributes: {
    'Gift wrap': 'Yes',
  },
  updates: {
    794864229: 1,
    794864101: 2,
  },
}).then((r) => console.log(r.note))
```
