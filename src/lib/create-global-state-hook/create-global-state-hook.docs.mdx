import createGlobalStateHook from '.'

export const meta = {
  type: 'function',
  name: 'createGlobalStateHook',
  description: 'Prints greeting to the console',
  tags: ['function'],
  lists: {
    arguments: [
      { name: 'name', type: 'string', description: 'Name that should be in the greeting' },
    ],
    returns: [
      { name: 'message', type: 'string', description: 'Text printed to the console' },
    ],
  },
  scope: { createGlobalStateHook },
}

## Playground

```jsx live render
const useUsername = createGlobalStateHook('')

const Greeting = () => {
  const [username] = useUsername()
  return <p>Welcome {username || 'Anonymous'}!</p>
}

const UserForm = () => {
  const [username, setUsername] = useUsername()
  return <input placeholder="What is your name?" value={username} onChange={e => setUsername(e.target.value)} />
}

const Page = () => (
  <div>
    <Greeting />
    <UserForm />
  </div>
)

render(<Page />)
```
