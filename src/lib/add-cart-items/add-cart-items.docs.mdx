## Usage

```js
import addCartItems from './add-cart-items'
addCartItems([
  {
    variantId: 123456,
    quantity: 2,
    properties: {
      discount: 0.6,
      shipping_discount: 0.3,
      create_bt_subscription: true,
      subscription_product_id: 123456,
    },
  },
  {
    variantId: 7889001,
  },
]).then((updatedProducts) =>
  console.log(`${updatedProducts.map((product) => product.title).join(' ')} now is in your cart`)
)
```
