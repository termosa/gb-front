import { ThemeProvider } from './theme-provider';
import styled from 'styled-components';

It requires Cormorant Garamond font to be available:
```
https://fonts.googleapis.com/css?family=Montserrat:400,500,600,700|Cormorant+Garamond:400,600,700,700i&display=swap
```

### Component usage
```jsx live
(
  Button = styled.button`
    background-color: ${({ theme }) => theme.colors.black};
    color: ${({ theme }) => theme.colors.white};
    padding: 1em;
    margin: 1em;
    border: none;
  `,
  <ThemeProvider>
    <Button>Button with themed styles</Button>
  </ThemeProvider>
)
```
