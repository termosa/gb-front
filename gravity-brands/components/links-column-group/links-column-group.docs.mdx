import { LinksColumnGroup } from './links-column-group'
import PresentationCanvas from '@gravity-brands/helpers.presentation-canvas'

Basic example:
```js live
<PresentationCanvas dark>
  <LinksColumnGroup
    title="Customer Support"
    links={[
      { name: 'Help Center', href: '#', onClick: e => e.preventDefault() },
      { name: 'Order Status', onClick: () => alert('Not available') }
    ]}
  />
</PresentationCanvas>
```
