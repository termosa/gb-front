import { Button } from './button'
import PresentationCanvas from '@fragrantjewels/gravity-brands.helpers.presentation-canvas'

Basic example:
```js live
<PresentationCanvas centered>
  <Button className="Component-Content">Content</Button>
</PresentationCanvas>
```

Custom colors:
```js live
<PresentationCanvas centered>
  <Button className="Component-Content" frontColor="#fff" backColor="#9059c8">Submit</Button>
</PresentationCanvas>
```
